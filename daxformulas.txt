Name	Calculation
COST	COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[PnL_TYPE]="Cost")
REVENUE	REVENUE:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[PnL_TYPE]="REVENUE")*-1
CGP	CGP:=([REVENUE]-[COST])/[REVENUE]
PROFIT	PROFIT:=[REVENUE]-[COST]
MULTIPLIER	MULTIPLIER:=[REVENUE]/[COST]
LABOR COST	LABOR COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="1B")
LABOR REVENUE	LABOR REVENUE:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="91")*-1
LABOR PROFIT	LABOR PROFIT:=[LABOR REVENUE]-[LABOR COST]
LGP	LGP:=([LABOR REVENUE]-[LABOR COST])/[LABOR REVENUE]
LABOR MULTIPLIER	LABOR MULTIPLIER:=[LABOR REVENUE]/[LABOR COST]
PPE COST	PPE COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="6S")
PPE REVENUE	PPE REVENUE:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="BS")*-1
PPE HR	PPE HR:=[PPE COST]/[LABOR HOURS]
LABOR HOURS	LABOR HOURS:=CALCULATE(SUM(JOB_COST_DETAIL[QUANTITY]),JOB_COST_DETAIL[COST_TYPE]="1B")
TOOLS COST	TOOLS COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="6T")
TOOLS HR	TOOLS HR:=[TOOLS COST]/[LABOR HOURS]
CONSUMABLE SCAFF COST	CONSUMABLE SCAFF COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="75")
CONSUMABLE HR SCAFF	CONSUMABLE HR SCAFF:=[CONSUMABLE SCAFF COST]/[LABOR HOURS]
CONSUMABLE NON SCAFF COST	CONSUMABLE NON SCAFF COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="7C")
CONSUMABLE HR NON SCAFF	CONSUMABLE HR NON SCAFF:=[CONSUMABLE NON SCAFF COST]/[LABOR HOURS]
TRUCK COSTS	TRUCK COSTS:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="TE")+CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="TR")+CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="TL")
TRUCK HR	TRUCK HR:=[TRUCK COSTS]/[LABOR HOURS]
JOB SITE OH COSTS	JOB SITE OH COSTS:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[COST_TYPE]="6B")
RENTAL REVENUE	RENTAL REVENUE:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[GLJC_TYPE]="Rental_Rev")*-1
RENTAL COST	RENTAL COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[GLJC_TYPE]="RENTAL_COST")
RGP	RGP:=([RENTAL REVENUE]-[RENTAL COST])/[RENTAL REVENUE]
SALES REVENUE	SALES REVENUE:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[GLJC_TYPE]="SALES_REV")*-1
SALES COST	SALES COST:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[GLJC_TYPE]="SALES_COST")
SGP	SGP:=([SALES REVENUE]-[SALES COST])/[SALES REVENUE]
RENTAL PROFIT	RENTAL PROFIT:=[RENTAL REVENUE]-[RENTAL COST]
SALES PROFIT	SALES PROFIT:=[SALES REVENUE]-[SALES COST]
RENTAL MULTIPLIER	RENTAL MULTIPLIER:=[RENTAL REVENUE]/[RENTAL COST]
SALES MULTIPLIER	SALES MULTIPLIER:=[SALES REVENUE]/[SALES COST]
Union Fringes	Union Fringes:=CALCULATE(SUM(JOB_COST_DETAIL[AMOUNT]),JOB_COST_DETAIL[JOB_ACCOUNT]="00000080")
Union Fringes Per HR	Union Fringes Per HR:=[Union Fringes]/[LABOR HOURS]
